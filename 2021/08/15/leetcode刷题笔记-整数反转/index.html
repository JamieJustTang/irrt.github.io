class Solution:
    def reverse(self, x: int) -> int:
        rev =int(str(x)[::-1] if x>=0 else ("-"+str(x)[:0:-1]))
        if len("{0:b}".format(rev if rev>=0 else -rev))>=32:
            return 0
        return rev
