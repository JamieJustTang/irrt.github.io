leetcode刷题笔记-最长公共后缀

2021-08-15
class Solution:
    def longestCommonPrefix(self, strs: List[str]) -> str:
        if len(strs)==1:
            return strs[0]
        for l in range(len(strs[0])):
            #for str in strs[1:]:
            for string in strs:
                try:
                    if strs[0][l]!=string[l]:
                        return strs[0][:l]
                except:
                    return strs[0][:l]
        return strs[0]
