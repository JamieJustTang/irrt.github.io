2021-09-15
class Solution:
    def romanToInt(self, s: str) -> int:
        num=0
        sdict={'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
        bydict={'IV':4,'IX':9,'XL':40,'XC':90,'CD':400,'CM':900}
        i=0
        while i < len(s)-1 :
            try:
                num+=bydict[s[i]+s[i+1]]
                i+=2
                #print(i,num)
            except KeyError:
                num+=sdict[s[i]]
                #print(i,num)
                # if i==len(s)-2:
                #     num+=sdict[s[i+1]]
                    #print(i+1,num)
                i+=1
            finally:
                continue
        if i==len(s)-1:
            num+=sdict[s[i]]
        return num

class Solution:
    def romanToInt(self, s: str) -> int:
        num=0
        sdict={'I':1,'V':5,'X':10,'L':50,'C':100,'D':500,'M':1000}
        bydict={'IV':-2,'IX':-2,'XL':-20,'XC':-20,'CD':-200,'CM':-200}
        for i in s:
            num+=sdict[i]
            #print(i,num)
        i=0
        while i < len(s)-1 :
            try:
                num+=bydict[s[i]+s[i+1]]
                i+=2
                #print(i,num)
            except:
                i+=1
            finally:
                continue

        return num
